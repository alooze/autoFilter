; Это конфигурационный файл для модуля

[html_names]
; В этой секции нужно задать сопоставление опций и имен полей в форме фильтрации
; Эта секция устарела и будет убрана в следующих версиях

;opt15 = phone
; Эта запись означает, что в форме будет поле name="phone", соответствующее опции с id=15
;
;optparent = category
; Эта запись означает, что в форме будет поле name="category", соответствующее опции parent


[html_aliases]
; В этой секции нужно задать сопоставление полных названий полей и алиасов

;opt10_select = sizes
; Эта запись означает, что название поля sizes должно читаться парсером форм, как opt10_select
